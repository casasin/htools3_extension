[docs]def addUnderline(sourceGlyph, targetGlyph, distanceY, thicknessY, bufferX, offsetX, clear=False, angle=0):
    '''
    Add an underline stroke with the given parameters to an existing glyph.

    Args:
        sourceGlyph (RGlyph): The input glyph to be underlined.
        targetGlyph (RGlyph): The destination glyph where the underlined shape will be drawn.
        distanceY (int): Distance to the bottom of the underline shape.
        thicknessY (int): Thickness of the underline.
        bufferX (int): Amount of units to exceed the glyph margins horizontally.
        offsetX (int): Amount of offset around contours to prevent overlapping shapes.
        clear (bool): Clear the contents of the target glyph before drawing.
        angle (float): The font's slant angle.

    .. code-block:: python

        sourceGlyph = CurrentGlyph()
        targetGlyph = sourceGlyph.getLayer('mask')
        angle = CurrentFont().info.italicAngle
        addUnderline(sourceGlyph, targetGlyph, 120, 80, 30, 30, clear=True, angle=angle)

    '''
    # make temp glyph for underline
    underlineGlyph = RGlyph()
    underlineGlyph.width = sourceGlyph.width
    pen = underlineGlyph.getPen()

    # draw plain underline
    x = -bufferX
    y = distanceY
    w = underlineGlyph.width + bufferX * 2
    h = thicknessY

    x1, y1 = x, y
    x2, y2 = x + w, y
    x3, y3 = x + w, y + h
    x4, y4 = x, y + h

    if angle:
        d = h * sin(radians(angle))
        x3 -= d
        x4 -= d

    pen.moveTo((x1, y1))
    pen.lineTo((x2, y2))
    pen.lineTo((x3, y3))
    pen.lineTo((x4, y4))
    pen.closePath()

    # apply slant offset
    if angle:
        f = sourceGlyph.font
        if f:
            slantOffset = tan(f.info.italicAngle * pi / 180.0) * (y - f.info.xHeight * 0.5)
            underlineGlyph.moveBy((-slantOffset, 0))

    # get intersections
    beamY = distanceY
    L = IntersectGlyphWithLine(sourceGlyph, ((-1000, beamY), (sourceGlyph.width + 1000, beamY)), canHaveComponent=True, addSideBearings=False)

    # clear overlaps
    if len(L):
        # make temp glyph for offset outline
        outlineGlyph = RGlyph()
        # expand outline
        sourceGlyphCopy = sourceGlyph.copy()
        sourceGlyphCopy.removeOverlap()
        expandGlyph(sourceGlyphCopy, outlineGlyph, offsetX, join=0, cap=0, inner=True, outer=False)
        # subtract offset from underline
        resultGlyph = underlineGlyph % outlineGlyph
        underlineGlyph = resultGlyph

    # draw result in target glyph
    targetGlyph.prepareUndo('add underline')
    if clear:
        targetGlyph.clear()
    targetGlyph.appendGlyph(underlineGlyph)
    targetGlyph.performUndo()
    targetGlyph.changed()



[docs]def addUnderlineFont(font, distanceY, thicknessY, bufferX, offsetX, exceptions=None):
    '''
    Add underline stroke to all glyphs in the font.

    Args:
        font (RFont): The input font with glyph contours to be underlined.
        distanceY (int): Distance to the bottom of the underline shape.
        thicknessY (int): Thickness of the underline.
        bufferX (int): Amount of units to exceed the glyph margins horizontally.
        offsetX (int): Amount of offset around contours to prevent overlapping shapes.
        exceptions (list): A list of glyph names which should **not** be underlined.

    .. code-block:: python

        f = OpenFont('path/to/font.ufo', showInterface=False)
        addUnderlineFont(f, 120, 80, 30, 30, exceptions=['underscore', 'space'])
        f.openInterface()

    '''
    # get italic angle
    a = font.info.italicAngle

    # add underline to glyphs
    for glyphName in font.keys():
        # ignore exceptions
        if glyphName in exceptions:
            continue
        # get source/target glyph
        sourceGlyph = font[glyphName]
        targetGlyph = sourceGlyph
        # decompose composed glyphs
        if len(sourceGlyph.components):
            sourceGlyph.decompose()
        # add underline to glyph
        addUnderline(sourceGlyph, targetGlyph, distanceY, thicknessY, bufferX, offsetX, clear=False, angle=a)
