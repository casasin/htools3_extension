    #: The scale of the glyph samples.

    #: Calculate scale to fit to page width or height.
    scaleToFit = False

    #: Toggle the display of glyph widths.

    #: Toggle the display of vertical metrics.

    glyphInfoColor = 0, 1, 1

    @property
    def glyphNames(self):
        '''
        The list of glyphs to be displayed in the sample.

        '''
        txt = self.text

        # handle simple strings
        if not '/' in txt:
            glyphNames = []
            for char in txt:
                uni = ord(char)
                glyphName = UV2AGL.get(uni)
                if glyphName:
                    glyphNames.append(glyphName)

        # parse gstring
        else:
            glyphNames = parseGString(self.font, txt)

        # handle missing glyphs
        if self.font:

            # replace missing glyphs with .notdef
            if '.notdef' in self.font:
                glyphNames = [glyphName if glyphName in self.font.keys() else '.notdef' for glyphName in glyphNames]

            # if the font has no .notdef glyph, remove missing glyph
            else:
                glyphNames = [glyphName for glyphName in glyphNames if glyphName in self.font.keys()]

        return glyphNames

    @property
    def parameters(self):
        '''
        The proof’s settings as a dictionary.

        '''
        attrs = [
            'text', 'layoutMode', 'layersAlpha',
            'scale', 'scaleToFit',
            'fontNameDraw', 'fontInfoFillColor',
            'glyphFillDraw', 'glyphFillColor',
            'glyphStrokeDraw', 'glyphStrokeColor', 'glyphStrokeWidth',
            'glyphInfoDraw', 'glyphInfoColor',
            'widthsDraw', 'widthsColor', 'widthsPadding',
            'vMetricsDraw', 'vMetricsColor',
            'anchorsDraw', 'anchorsColor', 'anchorsStrokeWidth', 'anchorsRadius',
            'boundsDraw', 'boundsColor',
            'wireframe', 'wireframeRadius', 'wireframeColor', 'wireframeStrokeWidth', 'wireframeDash',
            'guidesColor', 'guidesStrokeWidth', 'guidesDash',
        ]
        return {attr : getattr(self, attr) for attr in attrs}

        '''
        Draw the font’s vertical metrics.

        Args:
            guidesY (list): The font’s vertical metrics as a list of y-positions.

        '''
        ctx.save()

        if self.guidesDash:
            guidesDash = tuple([v * self.guidesStrokeWidth for v in self.guidesDash])
            ctx.lineDash(*guidesDash)
        else:
            ctx.lineDash(None)

        ctx.strokeWidth(self.guidesStrokeWidth)
        ctx.stroke(*self.vMetricsColor)

        for y in guidesY:
            if self.yBottom <= y <= self.yTop:
                ctx.line((self.xLeft, y), (self.xRight, y))

        ctx.restore()

        for x in guidesX:

                ctx.line((x, y1), (x, y2))

        ctx.restore()



[docs]    def drawGlyphInfo(self, glyph):
        '''
        Draw glyph info: glyph name, unicode, left margin, width, right margin. (single-glyph layout)

        Args:
            glyph (RGlyph): A glyph object.

        '''
        p = self.widthsPadding

        areaWidth, areaHeight = self.pageAreaWidth, self.pageAreaHeight

        w = areaWidth - p * 2
        h = self.captionLineHeight

        x1 = self.xLeft
        y1 = self.pageAreaBottom
        y2 = self.pageAreaTop - h

        uni = unicodeIntToHex(glyph.unicode)
        width, left, right = self.getGlyphMetrics(glyph)

        ctx.save()
        ctx.stroke(None)
        ctx.fill(*self.glyphInfoColor)
        ctx.font(self.captionFont)
        ctx.fontSize(self.captionFontSize)
        ctx.lineHeight(self.captionLineHeight)

        ctx.textBox(glyph.name, (x1, y2, w, h))
        if uni is not None:
            ctx.textBox(uni, (x1, y2, w, h), align='right')

        ctx.textBox(left,  (x1, y1, w, h), align='left')
        ctx.textBox(width, (x1, y1, w, h), align='center')
        ctx.textBox(right, (x1, y1, w, h), align='right')

        ctx.restore()



[docs]    def drawGlyphInfoMultiple(self, glyph, x, scale):
        '''
        Draw glyph info (multi-glyph layout).

        Args:
            glyph (RGlyph): A glyph object.
            x (float or int): The glyph’s horizontal position.
            scale (float): The glyph’s scale factor.

        '''
        h = self.captionLineHeight

        x1 = self.xLeft
        y1 = self.pageAreaBottom
        y2 = self.pageAreaTop - h

        p = self.widthsPadding

        if x < self.pageMargin:
            x1 = self.xLeft + p
        else:
            x1 = x + p

        right = x + glyph.width * scale

        if not right < self.pageMargin:
            glyphWidth = '%.2f' % glyph.width
            w = glyph.width * scale - p * 2
            ctx.save()
            ctx.fill(*self.glyphInfoColor)
            ctx.font(self.captionFont)
            ctx.fontSize(self.captionFontSize)
            ctx.textBox(glyph.name, (x1, y2, w, self.captionLineHeight))
            ctx.textBox(glyphWidth, (x1, y1, w, self.captionLineHeight))
            ctx.restore()



[docs]    def drawBounds(self, glyph, scale, pos):
        '''
        Draw glyph bounds.

        Args:
            glyph (RGlyph): A glyph object.
            pos (tuple): The glyph’s position as a tuple of (x,y) values.
            scale (float): The glyph’s scale factor.

        '''
        if not glyph.bounds:
            return

        x, y = pos
        L, B, R, T = glyph.bounds

        X = x + L * scale
        Y = y + B * scale
        W = (R - L) * scale
        H = (T - B) * scale

        ctx.save()
        ctx.stroke(None)
        ctx.fill(*self.boundsColor)
        ctx.rect(X, Y, W, H)
        ctx.restore()



[docs]    def drawAnchors(self, glyph, pos, scale):
        '''
        Draw all anchors in glyph.

        '''
        x, y = pos
        r1 = self.anchorsRadius
        r2 = r1 * 0.8

        ctx.save()
        ctx.translate(x, y)
        ctx.fill(None)
        ctx.stroke(*self.anchorsColor)
        ctx.strokeWidth(self.anchorsStrokeWidth)

        for a in glyph.anchors:
            X = a.x * scale
            Y = a.y * scale
            ctx.line((X - r1, Y), (X + r1, Y))
            ctx.line((X, Y - r1), (X, Y + r1))
            ctx.oval(X - r2, Y - r2, r2 * 2, r2 * 2)

        ctx.restore()



[docs]    def drawGlyph(self, glyph, scale):
        '''
        Draw a glyph with a given scale factor.

        Args:
            glyph (RGlyph): A glyph object.
            scale (float): The glyph’s scale factor.

        '''
        ctx.save()

        if self.wireframe:
            inverseScale = 1.0 / scale
            ctx.fill(None)
            ctx.stroke(*self.wireframeColor)
            ctx.strokeWidth(self.wireframeStrokeWidth * inverseScale)

        else:
            # set fill
            if self.glyphFillDraw and self.glyphFillColor is not None:
                if len(self.layers) > 1:
                    R, G, B = self.glyphFillColor[:3]
                    ctx.fill(R, G, B, self.layersAlpha)
                else:
                    ctx.fill(*self.glyphFillColor)
            else:
                ctx.fill(None)

            # set stroke
            if self.glyphStrokeDraw and self.glyphStrokeColor is not None:
                ctx.stroke(*self.glyphStrokeColor)
                ctx.strokeWidth(self.glyphStrokeWidth)
                ctx.lineCap('round')
                ctx.lineJoin('round')
            else:
                ctx.stroke(None)

        ctx.scale(scale)
        ctx.drawGlyph(glyph)
        ctx.restore()

        if self.wireframe:
            self.drawGlyphWireframe(glyph, scale)



[docs]    def drawGlyphWireframe(self, glyph, scale):
        '''
        Draw a glyph in wireframe mode.

        '''
        r = self.wireframeRadius

        ctx.save()
        ctx.fill(None)
        ctx.stroke(*self.wireframeColor)
        ctx.strokeWidth(self.wireframeStrokeWidth)

        if self.wireframeDash:
            wireframeDash = tuple([v * self.wireframeStrokeWidth for v in self.wireframeDash])
        else:
            wireframeDash = None

        for contour in glyph.contours:
            for i, pt in enumerate(contour.bPoints):

                # get bPoint position
                x1 = pt.anchor[0] * scale
                y1 = pt.anchor[1] * scale

                # draw incoming handle
                if pt.bcpIn[0] or pt.bcpIn[1]:
                    x2 = x1 + pt.bcpIn[0] * scale
                    y2 = y1 + pt.bcpIn[1] * scale
                    # handle
                    if wireframeDash:
                        ctx.lineDash(*wireframeDash)
                    ctx.line((x1, y1), (x2, y2))
                    # off-curve point
                    if wireframeDash:
                        ctx.lineDash(None)
                    ctx.oval(x2 - r, y2 - r, r*2, r*2)

                # draw outgoing handle
                if pt.bcpOut[0] or pt.bcpOut[1]:
                    x2 = x1 + pt.bcpOut[0] * scale
                    y2 = y1 + pt.bcpOut[1] * scale
                    # handle
                    if wireframeDash:
                        ctx.lineDash(*wireframeDash)
                    ctx.line((x1, y1), (x2, y2))
                    # off-curve point
                    if wireframeDash:
                        ctx.lineDash(None)
                    ctx.oval(x2 - r, y2 - r, r*2, r*2)

                # draw bPoint
                ctx.lineDash(None)
                ctx.oval(x1 - r, y1 - r, r*2, r*2)

        ctx.restore()

        if not len(self.glyphNames):

[docs]    def makeProofSingle(self):
        '''
        Make a PDF proof using the single glyph layout.

        '''
        pageWidth, pageHeight = self.pageSize
        areaWidth, areaHeight = self.pageAreaWidth, self.pageAreaHeight

        ctx.newDrawing()

        for i, glyphName in enumerate(self.glyphNames):

            # -----------
            # create page
            # -----------

            ctx.newPage(pageWidth, pageHeight)

            g = self.font[glyphName]

            # ---------------
            # calculate scale
            # ---------------

            if not self.scaleToFit:
                scale = self.scale

            # calculate scale-to-fit factors
            else:
                areaX, areaY = areaWidth, areaHeight

                if self.glyphInfoDraw:
                    areaY -= self.captionLineHeight * 4

                # get glyph box
                if g.bounds:
                    L, B, R, T = g.bounds
                    boxWidth  = R - L
                    boxHeight = T - B
                else:
                    boxWidth  = g.width
                    boxHeight = self.font.info.unitsPerEm * self.scale

                factorX = areaX  / float(boxWidth)
                factorY = areaY / float(boxHeight)

                # choose scale-to-fit factor
                if factorX <= factorY:
                    scale = factorX
                    xFit  = self.xLeft
                    yFit  = (pageHeight - (boxHeight * scale)) * 0.5
                else:
                    scale = factorY
                    xFit = (pageWidth - (boxWidth * scale)) * 0.5
                    yFit = self.pageAreaBottom

                if self.glyphInfoDraw:
                    yFit += self.captionLineHeight * 2

            # -----------------
            # get glyph metrics
            # -----------------

            w = g.width * scale
            h = self.font.info.unitsPerEm * scale
            d = abs(self.font.info.descender) * scale

            # calculate position

            if not self.scaleToFit:
                x = (pageWidth - w) * 0.5
                y = ((pageHeight - h) * 0.5) + abs(d)
            else:
                if g.bounds:
                    x = xFit - g.leftMargin * scale
                    y = yFit - B * scale
                else:
                    x = xFit
                    y = yFit

            # make guides
            guidesY = [
                y,
                y + self.font.info.xHeight   * scale,
                y + self.font.info.capHeight * scale,
                y + self.font.info.ascender  * scale,
                y - d,
            ]
            guidesX = [
                x,
                x + w,
            ]

            # ----------
            # draw glyph
            # ----------

            if self.fontNameDraw:
                # fontName = '%s %s' % (self.font.info.familyName, self.font.info.styleName)
                self.drawFontName()

            if self.boundsDraw:
                self.drawBounds(g, scale, (x, y))

            if self.widthsDraw:
                self.drawHorizontalMetrics(guidesX)

            if self.vMetricsDraw:
                self.drawVerticalMetrics(guidesY)

            if self.glyphInfoDraw:
                self.drawGlyphInfo(g)

            ctx.save()
            ctx.translate(x, y)
            self.drawGlyph(g, scale)
            ctx.restore()

            if self.anchorsDraw:
                self.drawAnchors(g, (x, y), scale)

            if self.pageGridDraw:
                self.drawPageGrid()

        # -----------------
        # draw page numbers
        # -----------------

        if self.pageDateTimeDraw:
            self.drawDateTime()

        if self.pageNumberDraw:
            self.drawPageNumbers()



[docs]    def makeProofMultiple(self):
        '''
        Make a PDF proof using the multi-glyph layout.

        '''
        pageWidth, pageHeight = self.pageSize
        areaWidth, areaHeight = self.pageAreaWidth, self.pageAreaHeight

        ctx.newDrawing()

        # -----------
        # create page
        # -----------

        ctx.newPage(pageWidth, pageHeight)

        # ---------------
        # calculate scale
        # ---------------

        if not self.scaleToFit:
            scale = self.scale

        # calculate scale-to-fit factors
        else:

            areaX, areaY = areaWidth, areaHeight

            if self.glyphInfoDraw:
                areaY -= self.captionLineHeight * 4

            # get line box
            lineL, lineB, lineR, lineT = [], [], [], []

            X = 0
            for glyphName in self.glyphNames:

                g = self.font[glyphName]

                # get glyph box
                if g.bounds:
                    L, B, R, T = g.bounds
                    lineL.append(L + X)
                    lineB.append(B)
                    lineR.append(R + X)
                    lineT.append(T)

                # handle empty glyphs
                else:
                    lineL.append(X)
                    lineB.append(abs(self.font.info.descender))
                    lineR.append(X + g.width)
                    lineT.append(self.font.info.unitsPerEm - abs(self.font.info.descender))

                # advance x
                X += g.width

            # calculate line size
            boxWidth  = max(lineR) - min(lineL)
            boxHeight = max(lineT) - min(lineB)

            factorX = areaX / float(boxWidth)
            factorY = areaY / float(boxHeight)

            # choose scale-to-fit factor
            if factorX <= factorY:
                scale = factorX
                xFit  = self.xLeft
                yFit  = (pageHeight - (boxHeight * scale)) * 0.5

            else:
                scale = factorY
                xFit  = (pageWidth - (boxWidth * scale)) * 0.5
                yFit  = self.pageAreaBottom

            if self.glyphInfoDraw:
                yFit += self.captionLineHeight * 2

        # ----------------
        # get line metrics
        # ----------------

        lineLength = sum([self.font[glyphName].width for glyphName in self.glyphNames])

        # get basic metrics
        w = lineLength * scale
        h = self.font.info.unitsPerEm * scale
        d = abs(self.font.info.descender) * scale

        # calculate position
        if not self.scaleToFit:
            x = (pageWidth - w) * 0.5
            y = ((pageHeight - h) * 0.5) + abs(d)
        else:
            x = xFit - self.font[self.glyphNames[0]].leftMargin * scale
            y = yFit - min(lineB) * scale

        # --------------------
        # draw bounds / widths
        # --------------------

        if self.boundsDraw or self.widthsDraw:

            X = x
            for glyphName in self.glyphNames:
                g = self.font[glyphName]

                if self.boundsDraw:
                    self.drawBounds(g, scale, (X, y))

                if self.widthsDraw:
                    guidesX = [
                        X,
                        X + g.width * scale,
                    ]
                    self.drawHorizontalMetrics(guidesX)

                X += g.width * scale

        # -------------
        # draw vmetrics
        # -------------

        if self.vMetricsDraw:
            guidesY = [
                y,
                y + self.font.info.xHeight   * scale,
                y + self.font.info.capHeight * scale,
                y + self.font.info.ascender  * scale,
                y - d,
            ]
            self.drawVerticalMetrics(guidesY)

        # -----------
        # draw glyphs
        # -----------

        X = x

        for glyphName in self.glyphNames:
            g = self.font[glyphName]

            ctx.save()
            ctx.translate(X, y)
            # ctx.blendMode('multiply')

            for i, layerName in enumerate(reversed(self.layers)):
                glyphLayer = g.getLayer(str(layerName))
                self.drawGlyph(glyphLayer, scale)

            ctx.restore()

            if self.glyphInfoDraw:
                self.drawGlyphInfoMultiple(g, X, scale)

            if self.anchorsDraw:
                self.drawAnchors(g, (X, y), scale)

            # done glyph
            X += g.width * scale

        # --------------
        # draw font name
        # --------------

        if self.fontNameDraw:
            self.drawFontName()

        if self.pageGridDraw:
            self.drawPageGrid()

        if self.pageDateTimeDraw:
            self.drawDateTime()

        if self.pageNumberDraw:
            self.drawPageNumbers()
