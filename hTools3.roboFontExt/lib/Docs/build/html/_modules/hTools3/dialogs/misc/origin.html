[docs]    def setPosition(self, posName):
        # clear positions
        if posName is None:
            for p in self.allPositions:
                checkbox = getattr(self, p)
                checkbox.set(False)
        # named position
        elif posName in self.allPositions:
            for p in self.allPositions:
                checkbox = getattr(self, p)
                if p == posName:
                    checkbox.set(True)
                else:
                    checkbox.set(False)
        # position not valid
        else:
            return


    @property
    def selected(self):
        for pos in self.allPositions:
            checkbox = getattr(self, pos)
            if checkbox.get():
                return pos

    @property
    def height(self):
        return self.getPosSize()[3]

    @property
    def width(self):
        return self.getPosSize()[4]

        for pos in self.allPositions:

                checkbox = getattr(self, pos)
                checkbox.set(False)

        if sender.get():

        else:
            if not self.allowEmpty:
                sender.set(True)
        self._callback(sender)
