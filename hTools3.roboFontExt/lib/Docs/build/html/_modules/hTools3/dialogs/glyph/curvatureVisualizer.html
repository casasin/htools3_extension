from __future__ import print_function

from vanilla import CheckBox, ColorWell, Slider, TextBox
from drawBot import BezierPath
from mojo import drawingTools as ctx
from hTools3.dialogs.glyphs.base import GlyphsDialogBase
from hTools3.modules.color import rgbToNSColor, nsColorToRGB
from hTools3.modules.curvatureVisualizer import CurvatureVisualizer


[docs]class CurvatureVisualizerDialog(GlyphsDialogBase):

    '''
    A dialog to visualize the curvature of bezier contours in the current glyph.

    >>> from hTools3.dialogs.glyph.curvatureVisualizer import CurvatureVisualizerDialog
    >>> CurvatureVisualizerDialog()

    '''

    title = 'curvature'
    key = '%s.curvatureVisualizer' % GlyphsDialogBase.key
    settings = {
        'curvatureCombDraw'  : True,
        'curvatureCombColor' : (0, 1, 0, 0.5),
        'curvatureCombSteps' : 20,
        'curvatureCombScale' : 1000,
    }

    def __init__(self):
        self.height  = self.textHeight * 6
        self.height += self.buttonHeight
        self.height += self.padding * 4
        self.w = self.window((self.width, self.height), self.title)

        x = p = self.padding
        y = p
        self.w.curvatureCombColor = ColorWell(
                (x, y, -p, self.buttonHeight),
                callback=self.updatePreviewCallback,
                color=rgbToNSColor(self.settings['curvatureCombColor']))

        y += self.buttonHeight + p * 0.75
        self.w.curvatureCombStepsLabel = TextBox(
                (x, y, -p, self.textHeight),
                "steps",
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.curvatureCombSteps = Slider(
                (x, y, -p, self.textHeight),
                value=self.settings['curvatureCombSteps'],
                minValue=10,
                maxValue=60,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p * 0.75
        self.w.curvatureCombScaleLabel = TextBox(
                (x, y, -p, self.textHeight),
                "scale",
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.curvatureCombScale = Slider(
                (x, y, -p, self.textHeight),
                value=self.settings['curvatureCombScale'],
                minValue=100,
                maxValue=2500,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p * 0.75
        self.w.preview = CheckBox(
                (x, y, -p, self.textHeight),
                "show preview",
                value=True,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.selectionOnly = CheckBox(
                (x, y, -p, self.textHeight),
                "selection only",
                value=False,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        self.initGlyphsWindowBehaviour()

        self.w.open()

    # -------------
    # dynamic attrs
    # -------------

    @property
    def curvatureCombSteps(self):
        return int(self.w.curvatureCombSteps.get())

    @property
    def curvatureCombScale(self):
        return self.w.curvatureCombScale.get()

    @property
    def curvatureCombColor(self):
        return nsColorToRGB(self.w.curvatureCombColor.get())

    @property
    def showPreview(self):
        return self.w.preview.get()

[docs]    def backgroundPreview(self, notification):

        g = notification['glyph']
        s = notification['scale']

        # -----------------
        # assert conditions
        # -----------------

        if not self.w.preview.get():
            return

        if g is None:
            return

        # ------------
        # draw preview
        # ------------

        parameters = {
            'curvatureCombDraw'         : self.showPreview,
            'curvatureCombColor'        : self.curvatureCombColor,
            'curvatureCombSteps'        : self.curvatureCombSteps,
            'curvatureCombScale'        : self.curvatureCombScale,
            'curvatureDraw'             : False,
            'handlesDraw'               : False,
            'tangentDraw'               : False,
            'curveDraw'                 : False,
            'pointsDraw'                : False,
            'linesAtTDraw'              : False,
            'curvatureCombStrokeWidth'  : 1 * s,
            'curvatureCombStrokeWidth2' : 2 * s,
        }

        B = BezierPath()
        # g.drawPoints(B)
        g.draw(B)

        V = CurvatureVisualizer(ctx)
        V.setParameters(parameters)
        V.draw(B)
