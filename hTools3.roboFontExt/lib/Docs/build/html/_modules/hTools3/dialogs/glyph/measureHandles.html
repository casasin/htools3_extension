[docs]class MeasureHandlesTool(GlyphsDialogBase):

    '''
    A drawing helper which shows measurements and angles of Bezier handles in the Glyph View.

    >>> from hTools3.dialogs.glyph.measureHandles import MeasureHandlesTool
    >>> MeasureHandlesTool()

    '''

    title = 'measure'
    key = '%s.measureHandles' % GlyphsDialogBase.key
    settings = {
        'handlesDraw'        : True,
        'handlesCaptionDraw' : True,
        'handlesColor'       : (1, 0, 0.5),
        'angleDraw'          : False,
        'boxDraw'            : False,
        'strokeAlpha'        : 0.5,
        'captionFontSize'    : 11,
    }

    def __init__(self):
        self.height  = self.textHeight * 7
        self.height += self.buttonHeight
        self.height += self.padding * 4 - 3
        self.w = self.window((self.width, self.height), self.title)

        x = p = self.padding
        y = p * 0.75
        self.w.handlesDraw = CheckBox(
                (x, y, -p, self.textHeight),
                "handles",
                value=self.settings['handlesDraw'],
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.angleDraw = CheckBox(
                (x, y, -p, self.textHeight),
                "angles",
                value=self.settings['angleDraw'],
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.boxDraw = CheckBox(
                (x, y, -p, self.textHeight),
                "projections",
                value=self.settings['boxDraw'],
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p * 0.75
        self.w.handlesColor = ColorWell(
                (x, y, -p, self.buttonHeight),
                callback=self.updatePreviewCallback,
                color=rgbToNSColor(self.settings['handlesColor']))

        y += self.buttonHeight + p * 0.75
        self.w.captionFontSizeLabel = TextBox(
                (x, y, -p, self.textHeight),
                "caption size",
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        valueMin  = 9
        valueMax  = 18
        tickCount = valueMax - valueMin
        self.w.captionFontSize = Slider(
                (x, y, -p, self.textHeight),
                value=self.settings['captionFontSize'],
                minValue=valueMin,
                maxValue=valueMax,
                tickMarkCount=tickCount,
                stopOnTickMarks=True,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p * 0.75
        self.w.preview = CheckBox(
                (x, y, -p, self.textHeight),
                "show preview",
                value=True,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.w.selectionOnly = CheckBox(
                (x, y, -p, self.textHeight),
                "selection only",
                value=False,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        self.measureHandles = MeasureHandles(ctx)

        self.initGlyphsWindowBehaviour()

        self.w.open()

[docs]    def backgroundPreview(self, notification):

        g = notification['glyph']
        s = notification['scale']

        # -----------------
        # assert conditions
        # -----------------

        if not self.w.preview.get():
            return

        if g is None:
            return

        # ------------
        # draw preview
        # ------------

        parameters = {
            'boxDraw'            : self.boxDraw,
            'boxColor'           : self.handlesColor,
            'boxCaptionDraw'     : False,
            'handlesDraw'        : self.handlesDraw,
            'handlesColor'       : self.handlesColor,
            'handlesCaptionDraw' : False,
            'angleDraw'          : self.angleDraw,
            'angleColor'         : self.handlesColor,
            'angleCaptionDraw'   : False,
            'strokeDash'         : 0, # self.strokeDash,
            'selectionOnly'      : self.selectionOnly,
        }
        self.measureHandles.setParameters(parameters)
        self.measureHandles.draw(g, scale=s)



[docs]    def foregroundPreview(self, notification):

        g = notification['glyph']
        s = notification['scale']

        # -----------------
        # assert conditions
        # -----------------

        if not self.w.preview.get():
            return

        if g is None:
            return

        # ------------
        # draw preview
        # ------------

        parameters = {
            'boxDraw'             : False,
            'boxCaptionDraw'      : self.boxDraw,
            'boxCaptionColor'     : self.handlesCaptionColor,
            'handlesDraw'         : False,
            'handlesCaptionDraw'  : self.handlesDraw,
            'handlesCaptionColor' : self.handlesCaptionColor,
            'angleDraw'           : False,
            'angleCaptionDraw'    : self.angleDraw,
            'angleCaptionColor'   : self.handlesCaptionColor,
            'selectionOnly'       : self.selectionOnly,
            'captionFontSize'     : self.captionFontSize,
            'captionFont'         : "LucidaGrande-Bold",
        }
        self.measureHandles.setParameters(parameters)
        self.measureHandles.draw(g, scale=s)
