    '''
    A dialog to make glyph proofs and generate PDFs.

    '''

    title = 'GlyphProofer'
    key = '%s.glyphProofer' % ProofDialogBase.key

    def __init__(self):
        self.height = 600
        self.width  = 960
        self.w = self.window((self.width, self.height), self.title, minSize=(self.width, self.height))

        # initialize proof
        self.proof = GlyphsProof()

        # initialize UI
        self.initFontSelectorGroup()
        self.initPageSelectorGroup()
        self.initLayoutSelectorGroup()
        self.initDisplaySelectorGroup()
        self.initSavingSelectorGroup()
        self.initSplitView()

        # initialize fonts & layers
        self.updateFonts()
        self.updateFontsList()
        self.loadLayers()

        # initialize window behavior
        self.setUpBaseWindowBehavior()
        addObserver(self, "updateCanvasObserver",    "currentGlyphChanged")
        addObserver(self, "updateFontsListObserver", "newFontDidOpen")
        addObserver(self, "updateFontsListObserver", "fontDidOpen")
        addObserver(self, "updateFontsListObserver", "fontDidClose")

        # initialize sample
        self.makeProof()
        self.updateCanvas()

        # open dialog
        self.w.open()

[docs]    def initLayoutSelectorGroup(self):
        '''
        Initialize layoutSelector group.

        '''
        self.layoutSelector = Group((0, 0, -0, -0))

        x = y = p = self.padding
        self.layoutSelector.glyphsMode = RadioGroup(
                (x, y, -p, self.textHeight * 2),

                sizeStyle=self.sizeStyle,
                callback=self.updateCanvasCallback)
        self.layoutSelector.glyphsMode.set(1)

        y += self.textHeight * 2 + p
        self.layoutSelector.textModeTitle = TextBox(
                (x, y, -p, self.textHeight),
                'sample text:',
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.layoutSelector.textMode = RadioGroup(
                (x, y, -p, self.textHeight * 2),
                ["font selection", "input text"],
                sizeStyle=self.sizeStyle,
                callback=self.updateCanvasCallback)
        self.layoutSelector.textMode.set(1)

        y += self.textHeight * 2 + p
        self.layoutSelector.sampleText = EditText(
                (x, y, -p, self.textHeight),
                'abcdefg',
                sizeStyle=self.sizeStyle,
                callback=self.updateCanvasCallback)

        y += self.textHeight + p
        self.layoutSelector.scaleTitle = TextBox(
                (x, y, -p, self.textHeight),
                'sample scale:',
                sizeStyle=self.sizeStyle)

        y += self.textHeight
        self.layoutSelector.scaleSlider = Slider(
                (x, y-5, -p, self.textHeight),
                value=0.2,
                minValue=0.01,
                maxValue=0.6,
                sizeStyle=self.sizeStyle,
                callback=self.updateCanvasCallback)
        self.layoutSelector.scaleSlider._nsObject.setVertical_(False)

        y += self.textHeight
        self.layoutSelector.fitToPage = CheckBox(
                (x, y, -p, self.textHeight),
                'fit to page',
                value=True,
                sizeStyle=self.sizeStyle,
                callback=self.updateCanvasCallback)

        y += self.textHeight + p
        self.layoutSelectorHeight = y

    @property
    def sampleText(self):
        # mode 0 : font selection
        if not self.layoutSelector.textMode.get():
            return '/%s' % '/'.join(self.font.selectedGlyphNames)
        # mode 1 : input string
        else:
            return self.layoutSelector.sampleText.get()

[docs]    def loadSettings(self, filePath):

        with open(filePath, 'r', encoding='utf-8') as inputFile:
            parameters = json.load(inputFile)

        # set proof parameters
        self.proof.setParameters(parameters)

        # update layout settings
        self.layoutSelector.glyphsMode.set(parameters['layoutMode'])
        self.layoutSelector.scaleSlider.set(parameters['scale'])
        self.layoutSelector.fitToPage.set(parameters['scaleToFit'])
        self.layoutSelector.sampleText.set(parameters['text'])

        # update display settings
        self.displaySelector.pageInfoDraw.set(parameters['fontNameDraw'])
        self.displaySelector.glyphInfoDraw.set(parameters['glyphInfoDraw'])
        self.displaySelector.glyphFillDraw.set(parameters['glyphFillDraw'])
        self.displaySelector.glyphStrokeDraw.set(parameters['glyphStrokeDraw'])
        self.displaySelector.glyphStrokeWidth.set(parameters['glyphStrokeWidth'])
        self.displaySelector.widthsDraw.set(parameters['widthsDraw'])
        self.displaySelector.vMetricsDraw.set(parameters['vMetricsDraw'])
        self.displaySelector.anchorsDraw.set(parameters['anchorsDraw'])
        self.displaySelector.boundsDraw.set(parameters['boundsDraw'])
        self.displaySelector.wireframe.set(parameters['wireframe'])

        # update color swatches
        self.displaySelector.pageInfoColor.set(rgbToNSColor(self.proof.captionFillColor))
        self.displaySelector.glyphInfoColor.set(rgbToNSColor(self.proof.glyphInfoColor))
        self.displaySelector.glyphFillColor.set(rgbToNSColor(self.proof.glyphFillColor))
        self.displaySelector.glyphStrokeColor.set(rgbToNSColor(self.proof.glyphStrokeColor))
        self.displaySelector.widthsColor.set(rgbToNSColor(self.proof.widthsColor))
        self.displaySelector.vMetricsColor.set(rgbToNSColor(self.proof.vMetricsColor))
        self.displaySelector.anchorsColor.set(rgbToNSColor(self.proof.anchorsColor))
        self.displaySelector.boundsColor.set(rgbToNSColor(self.proof.boundsColor))
        self.displaySelector.wireframeColor.set(rgbToNSColor(self.proof.wireframeColor))

        # update proof
        self.makeProof()
        self.updateCanvas()



[docs]    def makeProof(self):
        '''
        Make a PDF proof with the current settings.

        '''
        parameters = {
            'pageSize'         : self.pageSize,
            'font'             : self.font,
            'text'             : self.sampleText,
            'layers'           : self.layers,
            'layersAlpha'      : self.fontSelector.layersAlpha.get(),

            # layout
            'scale'            : self.layoutSelector.scaleSlider.get(),
            'scaleToFit'       : self.layoutSelector.fitToPage.get(),
            'layoutMode'       : self.layoutSelector.glyphsMode.get(),

            # display
            'fontNameDraw'     : self.displaySelector.pageInfoDraw.get(),
            'pageNumberDraw'   : self.displaySelector.pageInfoDraw.get(),
            'pageDateTimeDraw' : self.displaySelector.pageInfoDraw.get(),

            'glyphInfoDraw'    : self.displaySelector.glyphInfoDraw.get(),
            'glyphFillDraw'    : self.displaySelector.glyphFillDraw.get(),
            'glyphStrokeDraw'  : self.displaySelector.glyphStrokeDraw.get(),
            'glyphStrokeWidth' : self.displaySelector.glyphStrokeWidth.get(),
            'widthsDraw'       : self.displaySelector.widthsDraw.get(),
            'vMetricsDraw'     : self.displaySelector.vMetricsDraw.get(),
            'anchorsDraw'      : self.displaySelector.anchorsDraw.get(),
            'boundsDraw'       : self.displaySelector.boundsDraw.get(),
            'wireframe'        : self.displaySelector.wireframe.get(),

            # colors
            'captionFillColor' : self.pageInfoColor,
            'glyphInfoColor'   : self.glyphInfoColor,
            'glyphFillColor'   : self.glyphFillColor,
            'glyphStrokeColor' : self.glyphStrokeColor,
            'widthsColor'      : self.widthsColor,
            'vMetricsColor'    : self.vMetricsColor,
            'anchorsColor'     : self.anchorsColor,
            'boundsColor'      : self.boundsColor,
            'wireframeColor'   : self.wireframeColor,
        }
        self.proof.setParameters(parameters)
        self.proof.makeProof()
