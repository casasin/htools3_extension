    '''
    A dialog to add an underline stroke to the selected glyphs.

    .. code-block:: python

        from hTools3.dialogs.glyphs.underline import UnderlineGlyphsDialog

    title = 'underline'
    key = '%s.underline' % GlyphsDialogBase.key
    settings = {
        'distance'  : -100,
        'thickness' : 50,

    }

    def __init__(self):
        self.height  = self.spinnerHeight * 4
        self.height += self.buttonHeight
        self.height += self.padding * 9
        self.height += self.textHeight * 3
        self.w = self.window((self.width, self.height), self.title)

        x = y = p = self.padding
        self.w.distance = Spinner(
                (x, y),
                default=self.settings['distance'],
                integer=True,
                callback=self.updatePreviewCallback,
                label='position')

        y += self.w.distance.height + p
        self.w.distanceMode = RadioGroup(
                (x, y, -p * 0.5, self.textHeight),
                ["B", "M", "T"],
                isVertical=False,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)
        self.w.distanceMode.set(0)

        y += self.textHeight + p
        self.w.thickness = Spinner(
                (x, y),
                default=self.settings['thickness'],
                integer=True,
                callback=self.updatePreviewCallback,
                label='stroke')

        y += self.w.thickness.height + p
        self.w.slantOffset = CheckBox(
                (x, y, -p, self.textHeight),
                "slant offset",
                value=True,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p
        self.w.buffer = Spinner(
                (x, y),
                default=self.settings['buffer'],
                integer=True,
                callback=self.updatePreviewCallback,
                label='buffer')

        y += self.w.buffer.height + p
        self.w.offset = Spinner(
                (x, y),
                default=self.settings['offset'],
                integer=True,
                callback=self.updatePreviewCallback,
                label='margin')

        y += self.w.offset.height + p + 3
        self.w.buttonApply = SquareButton(
                (x, y, -p, self.buttonHeight),
                "apply",
                sizeStyle=self.sizeStyle,
                callback=self.applyCallback)

        y += self.buttonHeight + p
        self.w.preview = CheckBox(
                (x, y, -p, self.textHeight),
                "preview",
                value=True,
                callback=self.updatePreviewCallback,
                sizeStyle=self.sizeStyle)

        self.initGlyphsWindowBehaviour()
        self.w.open()

    @property
    def distanceMode(self):
        return self.w.distanceMode.get()

    @property
    def angle(self):
        angle = 0
        if self.slantOffset:
            font = CurrentFont()
            if font:
                angle = font.info.italicAngle
        return angle

    @property
    def d(self):
        # middle
        if self.distanceMode == 1:
            d = self.distance - self.thickness * 0.5
        # top
        elif self.distanceMode == 2:
            d = self.distance - self.thickness
        # bottom
        else:
            d = self.distance
        return d

[docs]    def backgroundPreview(self, notification):

        g = notification['glyph']
        s = notification['scale']

        if not self.w.preview.get():
            return

        if not g:
            return

        # ------------
        # make preview
        # ------------

        previewGlyph = self.makeGlyph(g, preview=True)

        # ------------
        # draw preview
        # ------------

        self.drawPreview(previewGlyph, s)

[docs]    def drawPreview(self, glyph, previewScale):

        ctx.save()

        # draw underline preview
        ctx.fill(*self.previewFillColor)
        ctx.stroke(*self.previewStrokeColor)
        ctx.strokeWidth(self.previewStrokeWidth * previewScale)
        ctx.drawGlyph(glyph)

        # draw position line
        y  = self.distance
        x1 = -10000
        x2 =  10000
        ctx.lineDash(self.previewStrokeWidth * previewScale, self.previewStrokeWidth * previewScale)
        ctx.line((x1, y), (x2, y))

        # done
        ctx.restore()



[docs]    def makeGlyph(self, glyph, preview=False):
        if preview:
            dstGlyph = glyph.copy()
        else:
            dstGlyph = glyph
        # add underline
        addUnderline(glyph, dstGlyph, self.d, self.thickness, self.buffer, self.offset, clear=False, angle=self.angle)
        # done
        return dstGlyph



[docs]    def apply(self):

        # -----------------
        # assert conditions
        # -----------------

        font = self.getCurrentFont()
        if not font:
            return

        glyphNames = self.getGlyphNames()
        if not glyphNames:
            return

        # ----------
        # print info
        # ----------

        print('adding underline to glyphs:\n')
        print('\tdistance: %s' % self.distance)
        print('\tthickness: %s' % self.thickness)
        print('\tbuffer: %s' % self.buffer)
        print('\tmargin: %s' % self.offset)
        print()
        print('\t', end='')
        print(' '.join(glyphNames), end='\n')

        # -----------------
        # transform glyphs
        # -----------------

        for glyphName in glyphNames:
            g = font[glyphName]
            g.prepareUndo('add underline')
            self.makeGlyph(g)
            g.performUndo()
            g.changed()

        # done
        font.changed()
        print('\n...done.\n')
