[docs]    def backgroundPreview(self, notification):
        g = notification['glyph']
        s = notification['scale']

        #-------------------
        # assert conditions
        #-------------------

        if not self.w.preview.get():
            return

        if not g:
            return

        if not g.bounds:
            return

        #--------------
        # make preview
        #--------------

        previewGlyph = self.makeGlyph(g, preview=True)

        #--------------
        # draw preview
        #--------------

        self.drawPreview(previewGlyph, s)

        font = self.getCurrentFont()

            self.w.sourceLayer.setItems(font.layerOrder)



[docs]    def updateTargetLayer(self):
        font = self.getCurrentFont()
        if not font:
            self.w.targetLayer.setItems([])
        else:
            targetLayers = list(font.layerOrder)
            targetLayers.remove(self.sourceLayer)
            self.w.targetLayer.setItems(targetLayers)



[docs]    def drawPreview(self, glyph, previewScale):
        ctx.save()
        ctx.fill(*self.previewFillColor)
        ctx.stroke(*self.previewStrokeColor)
        ctx.strokeWidth(self.previewStrokeWidth * previewScale)
        ctx.drawGlyph(glyph)
        ctx.restore()



[docs]    def makeGlyph(self, glyph, preview=False):

        # correct src glyph direction
        srcGlyph = glyph.getLayer(self.sourceLayer).copy()
        srcGlyph.correctDirection()

        # get dest glyph
        if preview:
            dstGlyph = RGlyph()
        else:
            dstGlyph = glyph.getLayer(self.targetLayer)

        # apply outline
        # NOTE: inner and outer are reversed (??)
        expandGlyph(srcGlyph, dstGlyph, self.distance, join=self.join, cap=self.cap, inner=self.outer, outer=self.inner)

        # done
        return dstGlyph



[docs]    def apply(self):

        # -----------------
        # assert conditions
        # -----------------

        font = self.getCurrentFont()
        if not font:
            return

        glyphNames = self.getGlyphNames()
        if not glyphNames:
            return

        # ----------
        # print info
        # ----------

        if self.verbose:
            print('outlining contours...\n')
            print('\tdistance: %s' % self.distance)
            print('\tjoin: %s'     % self.strokeParameters[self.join])
            print('\tcap: %s'      % self.strokeParameters[self.cap])
            print('\tinner: %s'    % bool(self.inner))
            print('\touter: %s'    % bool(self.outer))
            print()
            print('\t', end='')
            print(' '.join(glyphNames), end='\n')

        # ----------------
        # transform glyphs
        # ----------------

        for glyphName in glyphNames:
            g = font[glyphName]
            g.prepareUndo('outline glyphs')
            self.makeGlyph(g)
            g.changed()
            g.performUndo()

        # done
        font.changed()
        print('\n...done.\n')
