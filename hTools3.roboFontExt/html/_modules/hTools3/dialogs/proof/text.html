    @property
    def psFontName(self):
        if not self.font:
            return
        fontName = self.font.info.postscriptFontName
        if fontName is None:
            fontName = '%s-%s' % (self.font.info.familyName, self.font.info.styleName)
            fontName = fontName.replace(' ', '')
        return fontName

    @property
    def textAlignment(self):
        selection = self.textSampleSelector.textAlignment.get()
        return ['left', 'center', 'right'][selection]

    @property
    def selectedTextSamples(self):

        if not len(self.textPropertiesSelector.textSamplesList.get()):
            return

        selection = self.textPropertiesSelector.textSamplesList.getSelection()
        if not len(selection):
            return

        sampleNames = self.textPropertiesSelector.textSamplesList.get()

        D = dict()
        for i, sampleName in enumerate(sampleNames):
            if i in selection:
                D[sampleName] = self.textSamples[sampleName]

        return D

    @property
    def fontFeatures(self):
        if self.font is None:
            return []
        fontFeaturesDict = ctx.listOpenTypeFeatures(self.psFontName)
        return list(fontFeaturesDict.keys())

    @property
    def selectedFontFeatures(self):
        fontFeatures = self.fontFeaturesSelector.featuresList.get()
        selection    = self.fontFeaturesSelector.featuresList.getSelection()
        return [fea for i, fea in enumerate(fontFeatures) if i in selection]

    @property

[docs]    def editSampleTextNameCallback(self, sender):

        listItems = self.sheet.textSamplesList.get()
        namesNew  = set([listItem['text'] for listItem in listItems])
        namesOld  = set(self.textSamples.keys())

        newName = namesNew.difference(namesOld)
        oldName = namesOld.difference(namesNew)

        if not (len(newName) and len(oldName)):
            return

        newName = list(newName)[0]
        oldName = list(oldName)[0]

        # rename key in internal dict
        textSamples = dict()
        for k, v in self.textSamples.items():
            if k == oldName:
                textSamples[newName] = v
            else:
                textSamples[k] = v

        self.textSamples = textSamples

[docs]    def loadSettings(self, filePath):

        with open(filePath, 'r', encoding='utf-8') as inputFile:
            parameters = json.load(inputFile)

        textAlignmentOptions = ['left', 'center', 'right']
        textAlignmentIndex = textAlignmentOptions.index(parameters['textAlignment'])

        # set proof parameters
        self.proof.setParameters(parameters)

        # update layout settings
        self.textSampleSelector.fontSize.set(parameters['textFontSize'])
        self.textSampleSelector.tracking.set(parameters['textTracking'])
        self.textSampleSelector.lineHeight.set(parameters['textLineHeight'])
        self.textSampleSelector.textAlignment.set(textAlignmentIndex)
        self.textSampleSelector.hyphenation.set(parameters['textHyphenation'])

        # update display settings
        self.displaySelector.pageInfoDraw.set(parameters['fontNameDraw'])
        self.displaySelector.textFillDraw.set(parameters['textFillDraw'])
        self.displaySelector.textStrokeDraw.set(parameters['textStrokeDraw'])
        self.displaySelector.vMetricsDraw.set(parameters['vMetricsDraw'])
        self.displaySelector.textStrokeWidth.set(parameters['textStrokeWidth'])

        # update color swatches
        self.displaySelector.pageInfoColor.set(rgbToNSColor(self.proof.captionFillColor))
        self.displaySelector.textFillColor.set(rgbToNSColor(self.proof.textFillColor))
        self.displaySelector.textStrokeColor.set(rgbToNSColor(self.proof.textStrokeColor))
        self.displaySelector.vMetricsColor.set(rgbToNSColor(self.proof.vMetricsColor))

        # load sample texts
        self.textSamples = parameters['sampleTexts']
        self.updateTextSamplesList()

        # update proof
        # self.updateFontFeaturesList()
        self.makeProof()
        self.updateCanvas()



[docs]    def makeProof(self):

        if not self.psFontName:
            msg = 'no font provided'
            # Message(msg, title=self.title, informativeText='')
            print(msg)
            return

        if not self.psFontName in ctx.installedFonts():
            msg  = '%s not installed.\n' % self.psFontName
            msg += 'Please test install the font before making a proof.'
            # Message(msg, title=self.title, informativeText='')
            print(msg)
            return

        parameters = {
            'pageSize'            : self.pageSize,
            'pageGridDraw'        : False,
            'pageNumberDraw'      : self.displaySelector.pageInfoDraw.get(),
            'pageDateTimeDraw'    : self.displaySelector.pageInfoDraw.get(),
            'fontNameDraw'        : self.displaySelector.pageInfoDraw.get(),
            'sampleNameDraw'      : self.displaySelector.pageInfoDraw.get(),
            'captionFillColor'    : self.pageInfoColor,
            'textFillDraw'        : self.displaySelector.textFillDraw.get(),
            'textFillColor'       : self.textFillColor,
            'textStrokeDraw'      : self.displaySelector.textStrokeDraw.get(),
            'textStrokeColor'     : self.textStrokeColor,
            'textStrokeWidth'     : self.displaySelector.textStrokeWidth.get(),
            'vMetricsDraw'        : self.displaySelector.vMetricsDraw.get(),
            'vMetricsColor'       : self.vMetricsColor,
            'vMetricsStrokeWidth' : 1,
            'font'                : self.font,
            'psFontName'          : self.psFontName,
            'fontFeatures'        : self.selectedFontFeatures,
            'sampleTexts'         : self.selectedTextSamples,
            'textFontSize'        : self.textSampleSelector.fontSize.get(),
            'textLineHeight'      : self.textSampleSelector.lineHeight.get(),
            'textTracking'        : self.textSampleSelector.tracking.get(),
            'textAlignment'       : self.textAlignment,
            'textHyphenation'     : self.textSampleSelector.hyphenation.get(),
        }
        self.proof.setParameters(parameters)
        self.proof.makeProof()
