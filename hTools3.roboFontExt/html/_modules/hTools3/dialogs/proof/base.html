    @property
    def font(self):
        '''
        Selected font.

        Returns:
            A font object (RFont).

        '''
        allFonts = self.fontSelector.allFonts.getItems()
        if not len(allFonts):
            return
        fontName = allFonts[self.fontSelector.allFonts.get()]
        return self.allFonts[fontName]

    @property
    def layers(self):
        '''
        Selected layers.

        Returns:
            A list of layer names.

        '''
        fontLayers = self.fontSelector.layersList.get()
        selection  = self.fontSelector.layersList.getSelection()
        return [fontLayers[i] for i in selection]

    @property
    def pageSize(self):
        '''
        The selected page size.

        Returns:
            A tuple with page width and height.

        '''
        customPage = self.pageSelector.customPage.get()

        if not customPage:
            pageSize = self.pageSizes[self.pageSelector.pageSizes.get()]
            if self.pageSelector.landscape.get():
                pageSize += 'Landscape'

        else:
            pageW = int(self.pageSelector.pageWidth.get())
            pageH = int(self.pageSelector.pageHeight.get())
            pageUnits = self.pageSelector.pageUnits.get()

            # convert mm to pts
            if pageUnits == 0:
                pageW = mm2pt(pageW)
                pageH = mm2pt(pageH)

            # convert inches to pts
            elif pageUnits == 1:
                pageW = pageW * 72
                pageH = pageH * 72

            pageSize = pageW, pageH

        return pageSize

        msg = 'import proof settings from .json file'
        filePath = GetFile(message=msg, fileTypes=['json'])
        if not filePath:

[docs]    def updateFonts(self):
        '''
        Update the internal fonts dict.

        '''
        self.allFonts = {}
        for f in AllFonts():
            fontName = '%s %s' % (f.info.familyName, f.info.styleName)
            self.allFonts[fontName] = f
