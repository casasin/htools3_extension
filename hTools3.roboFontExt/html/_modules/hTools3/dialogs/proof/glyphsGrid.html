    def __init__(self):
        self.height = 600
        self.width  = 960
        self.w = self.window((self.width, self.height), self.title, minSize=(self.width, self.height))

        # initialize proof
        self.proof = GlyphsGridProof()

        # initialize UI
        self.initFontSelectorGroup()
        self.initPageSelectorGroup()
        self.initLayoutSelectorGroup()
        self.initDisplaySelectorGroup()
        self.initSavingSelectorGroup()
        self.initSplitView()

        # initialize fonts & layers
        self.updateFonts()
        self.updateFontsList()
        self.loadLayers()

        # initialize window behavior
        self.setUpBaseWindowBehavior()
        addObserver(self, "updateCanvasObserver",    "currentGlyphChanged")
        addObserver(self, "updateFontsListObserver", "newFontDidOpen")
        addObserver(self, "updateFontsListObserver", "fontDidOpen")
        addObserver(self, "updateFontsListObserver", "fontDidClose")

        # initialize sample
        self.makeProof()
        self.updateCanvas()

        # open dialog
        self.w.open()

[docs]    def initLayoutSelectorGroup(self):

        self.layoutSelector = Group((0, 0, -0, -0))

        x = y = p = self.padding
        self.layoutSelector.cellSizeLabel = TextBox(
                (x, y, -p, self.textHeight),
                'cell size:',
                sizeStyle=self.sizeStyle)
        y += self.textHeight
        self.layoutSelector.cellSize = Slider(
                (x, y, -p, self.textHeight),
                value=self.proof.cellSize,
                minValue=self.proof.cellSize * 0.5,
                maxValue=self.proof.cellSize * 3,
                callback=self.updateCanvasCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p
        self.layoutSelector.cellMarginLabel = TextBox(
                (x, y, -p, self.textHeight),
                'cell margin:',
                sizeStyle=self.sizeStyle)
        y += self.textHeight
        self.layoutSelector.cellMargin = Slider(
                (x, y, -p, self.textHeight),
                value=self.proof.cellMargin,
                minValue=0,
                maxValue=self.proof.cellMargin * 3,
                callback=self.updateCanvasCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p
        self.layoutSelector.glyphScaleLabel = TextBox(
                (x, y, -p, self.textHeight),
                'glyph scale:',
                sizeStyle=self.sizeStyle)
        y += self.textHeight
        self.layoutSelector.glyphScale = Slider(
                (x, y, -p, self.textHeight),
                value=self.proof.glyphScale,
                minValue=0.01,
                maxValue=1.0,
                callback=self.updateCanvasCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p * 0.5
        self.layoutSelector.cellSizeFit = CheckBox(
                (x, y, -p, self.textHeight),
                'fit to page',
                value=False,
                callback=self.updateCanvasCallback,
                sizeStyle=self.sizeStyle)

        y += self.textHeight + p
        self.layoutSelectorHeight = y

[docs]    def loadSettings(self, filePath):

        with open(filePath, 'r', encoding='utf-8') as inputFile:
            parameters = json.load(inputFile)

        # set proof parameters
        self.proof.setParameters(parameters)

        # update layout settings
        self.layoutSelector.glyphScale.set(parameters['glyphScale'])
        self.layoutSelector.cellSize.set(parameters['cellSize'])
        self.layoutSelector.cellMargin.set(parameters['cellMargin'])
        self.layoutSelector.cellSizeFit.set(parameters['cellSizeFit'])

        # update display settings
        self.displaySelector.pageInfoDraw.set(parameters['fontNameDraw'])
        self.displaySelector.glyphInfoDraw.set(parameters['glyphNameDraw'])
        self.displaySelector.glyphFillDraw.set(parameters['glyphFillDraw'])
        self.displaySelector.glyphStrokeDraw.set(parameters['glyphStrokeDraw'])
        self.displaySelector.vMetricsDraw.set(parameters['vMetricsDraw'])
        self.displaySelector.marginsDraw.set(parameters['marginsDraw'])
        self.displaySelector.cellFillDraw.set(parameters['cellFillDraw'])
        self.displaySelector.cellStrokeDraw.set(parameters['cellStrokeDraw'])

        # update color swatches
        self.displaySelector.pageInfoColor.set(rgbToNSColor(self.proof.captionFillColor))
        self.displaySelector.glyphInfoColor.set(rgbToNSColor(self.proof.cellCaptionColor))
        self.displaySelector.glyphFillColor.set(rgbToNSColor(self.proof.glyphFillColor))
        self.displaySelector.glyphStrokeColor.set(rgbToNSColor(self.proof.glyphStrokeColor))
        self.displaySelector.vMetricsColor.set(rgbToNSColor(self.proof.vMetricsColor))
        self.displaySelector.marginsColor.set(rgbToNSColor(self.proof.marginsColor))
        self.displaySelector.cellFillColor.set(rgbToNSColor(self.proof.cellFillColor))
        self.displaySelector.cellStrokeColor.set(rgbToNSColor(self.proof.cellStrokeColor))

        # update proof
        self.makeProof()
        self.updateCanvas()



[docs]    def makeProof(self, pageAmount=1):
        parameters = {
            'pageSize'          : self.pageSize,
            'pageAmount'        : pageAmount,
            'font'              : self.font,
            'layers'            : self.layers,
            'layersAlpha'       : self.fontSelector.layersAlpha.get(),

            # layout
            'cellSize'          : self.layoutSelector.cellSize.get(),
            'cellMargin'        : self.layoutSelector.cellMargin.get(),
            'cellSizeFit'       : self.layoutSelector.cellSizeFit.get(),
            'glyphScale'        : self.layoutSelector.glyphScale.get(),

            # display
            'fontNameDraw'      : self.displaySelector.pageInfoDraw.get(),
            'pageNumberDraw'    : self.displaySelector.pageInfoDraw.get(),
            'pageDateTimeDraw'  : self.displaySelector.pageInfoDraw.get(),
            'pageDateTimeDraw'  : self.displaySelector.pageInfoDraw.get(),

            'glyphNameDraw'     : self.displaySelector.glyphInfoDraw.get(),
            'glyphUnicodeDraw'  : self.displaySelector.glyphInfoDraw.get(),
            'glyphWidthDraw'    : self.displaySelector.glyphInfoDraw.get(),
            'glyphMarginsDraw'  : self.displaySelector.glyphInfoDraw.get(),

            'glyphFillDraw'     : self.displaySelector.glyphFillDraw.get(),
            'glyphStrokeDraw'   : self.displaySelector.glyphStrokeDraw.get(),
            'vMetricsDraw'      : self.displaySelector.vMetricsDraw.get(),
            'marginsDraw'       : self.displaySelector.marginsDraw.get(),
            'cellFillDraw'      : self.displaySelector.cellFillDraw.get(),
            'cellStrokeDraw'    : self.displaySelector.cellStrokeDraw.get(),

            # colors
            'captionFillColor'  : self.pageInfoColor,
            'cellCaptionColor'  : self.glyphInfoColor,
            'glyphFillColor'    : self.glyphFillColor,
            'glyphStrokeColor'  : self.glyphStrokeColor,
            'vMetricsColor'     : self.vMetricsColor,
            'marginsColor'      : self.marginsColor,
            'cellFillColor'     : self.cellFillColor,
            'cellStrokeColor'   : self.cellStrokeColor,

        }
        self.proof.setParameters(parameters)
        self.proof.makeProof()
