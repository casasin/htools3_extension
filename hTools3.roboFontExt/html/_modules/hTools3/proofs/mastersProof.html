[docs]    def makeProof(self):
        '''
        Make a PDF proof with the current settings.

        '''
        pageWidth, pageHeight = self.pageSize
        s = self.glyphScale

        G = GlyphCell()
        G.setParameters(self.glyphCellParameters)

        ctx.newDrawing()

        for glyphName in self.glyphNames:

            ctx.newPage(pageWidth, pageHeight)
            ctx.save()

            x = self.xLeft
            y = self.pageAreaBottom

            if self.pageGridDraw:
                self.drawPageGrid()

            if self.fontNameDraw:
                self.drawFontName()

            contours = []
            for font in self.fontsList:
                contours.append(len(font[glyphName]))
            G.contoursMax = max(contours)

            for i, font in enumerate(self.fontsList):

                G.glyph  = font[glyphName]

                G.width  = G.glyph.width * s
                G.height = G.glyph.font.info.unitsPerEm * s

                yDescender = abs(G.glyph.font.info.descender)
                yAscender  = G.glyph.font.info.ascender
                lineGap    = (G.glyph.font.info.unitsPerEm - (yAscender + abs(G.glyph.font.info.descender))) * 0.5

                # baseline
                yGapBottom  = (yDescender + lineGap) * s
                yDistance   = (self.pageAreaHeight - G.glyph.font.info.unitsPerEm * s) * 0.5
                yCellBottom = y + yDistance
                yBaseline   = yCellBottom + yGapBottom

                G.draw((x, yCellBottom), yBaseline)

                # connect points

                r = 1.5

                if self.showCompatiblePoints:
                    metricsDash = tuple([v * self.glyphMetricsStrokeWidth for v in self.glyphMetricsDash])

                    g1 = font[glyphName]

                    if i < len(self.fontsList) - 1:
                        g2 = self.fontsList[i+1][glyphName]

                    ctx.save()
                    ctx.translate(x, yCellBottom)

                    for ci, contour in enumerate(g1.contours):

                        color = G.contourColors[ci]

                        pt1 = g1[ci].points[0]

                        x1 = pt1.x * s
                        y1 = pt1.y * s + yGapBottom

                        if i < len(self.fontsList) - 1:

                            pt2 = g2[ci].points[0]

                            x2 = pt2.x * s + g1.width * s + self.glyphCellMargin
                            y2 = pt2.y * s + yGapBottom

                            ctx.stroke(*color)
                            ctx.fill(None)
                            ctx.lineDash(*metricsDash)
                            ctx.line((x1, y1), (x2, y2))

                        ctx.fill(*color)
                        ctx.stroke(None)
                        ctx.oval(x1-r, y1-r, r*2, r*2)

                    ctx.restore()

                    g1 = g2

                # done with font
                x += G.width + self.glyphCellMargin

            # done with glyph
            ctx.restore()

        # done with pages
        if self.pageNumberDraw:
            self.drawPageNumbers()

        if self.pageDateTimeDraw:
            self.drawDateTime()
