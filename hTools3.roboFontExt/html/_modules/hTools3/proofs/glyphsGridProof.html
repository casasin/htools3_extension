[docs]    def getCellSize(self):
        '''
        Get the cell size based on the current settings.

        Returns:
            The size of the glyph cells (as a float or int).

        '''
        if not self.cellSizeFit:
            return self.cellSize

        # fit glyph cells to page
        else:
            pageWidth, pageHeight = self.pageSize
            x = self.pageMargin
            xRight = pageWidth - self.pageMargin - self.cellSize
            cols = 0

            while x < xRight:
                x += self.cellSize + self.cellMargin
                cols += 1

            cellSize = (pageWidth - self.pageMargin*2 - self.cellMargin * (cols-1)) / cols

            return cellSize



[docs]    def getGlyphPosScale(self, glyph, pos, cellSize, unitsPerEm):
        '''
        Get the glyph position and scale based on the current settings.

        Returns:
            The position and scale for the glyph.

        '''
        x, y = pos
        bounds = glyph.bounds

        # calculate scale-to-fit factors
        if self.glyphScaleToFit:

            if not bounds:
                return

            L, B, R, T = bounds
            boxWidth  = R - L
            boxHeight = T - B

            factorX = cellSize / float(boxWidth)
            factorY = cellSize / float(boxHeight)

            if factorX <= factorY:
                scale = factorX
                X = x - L * scale
                Y = y + (cellSize - boxHeight * scale) * 0.5 - B * scale
            else:
                scale = factorY
                X = x + (cellSize - boxWidth * scale) * 0.5 - L * scale
                Y = y - B * scale

        # use fixed scale for all glyphs
        else:
            scale = cellSize / unitsPerEm * self.glyphScale
            X = x + (cellSize - glyph.width * scale) * 0.5
            Y = y + cellSize * self.glyphBaselineFactor

        return (X, Y), scale

[docs]    def drawGlyphInfo(self, glyph, pos, cellSize):
        '''
        Draw the glyph info: glyph name, unicode, width, left and right margins.

        Args:
            glyph (RGlyph): A glyph object.
            pos (tuple): The position of the glyph cell as a tuple of (x,y) values.
            cellSize (tuple): The size of the glyph cell as a tuple of width and height.

        '''
        x, y = pos

        w = cellSize - self.cellPadding * 2
        h = self.cellCaptionSize

        x1 = x + self.cellPadding
        y1 = y + self.cellPadding
        y2 = y + cellSize - self.cellPadding - self.cellCaptionSize

        width, leftMargin, rightMargin = self.getGlyphMetrics(glyph)

        ctx.save()
        ctx.fill(*self.cellCaptionColor)
        ctx.fontSize(self.cellCaptionSize)
        ctx.lineHeight(self.cellCaptionSize)
        ctx.font(self.captionFont)

        if self.glyphNameDraw:
            ctx.textBox(glyph.name, (x1, y2, w, h), align='left')

        if self.glyphUnicodeDraw and glyph.unicode:
            glyphUnicode = unicodeIntToHex(glyph.unicode)
            ctx.textBox(glyphUnicode, (x1, y2, w, h), align='right')

        if self.glyphWidthDraw and width is not None:
            ctx.textBox(width, (x1, y1, w, h), align='center')

        if self.glyphMarginsDraw:
            if leftMargin is not None:
                ctx.textBox(leftMargin, (x1, y1, w, h), align='left')
            if rightMargin is not None:
                ctx.textBox(rightMargin, (x1, y1, w, h), align='right')

        ctx.restore()



[docs]    def drawGlyphMargins(self, glyph, posCell, posGlyph, cellSize, scale):
        '''
        Draw the glyph margins.

        Args:
            glyph (RGlyph): A glyph object.
            posCell (tuple): The position of the glyph cell as a tuple of (x,y) values.
            posGlyph (tuple): The position of the glyph as a tuple of (x,y) values.
            cellSize (tuple): The size of the glyph cell as a tuple of width and height.
            scale (float): The scale factor of the glyph.

        '''
        x0, y0 = posCell
        x,  y  = posGlyph

        xValues = [
            x,
            x + glyph.width * scale,
        ]

        ctx.save()
        ctx.stroke(*self.marginsColor)
        ctx.strokeWidth(self.marginsStrokeWidth)
        self._setLineDash(self.marginsDash, self.marginsStrokeWidth)

        for x in xValues:
            ctx.line((x, y0), (x, y0 + cellSize))

        ctx.restore()



[docs]    def drawGlyphVMetrics(self, glyph, posCell, posGlyph, cellSize, scale):
        '''
        Draw the vertical metrics for a cell.

        Args:
            glyph (RGlyph): A glyph object.
            posCell (tuple): The position of the glyph cell as a tuple of (x,y) values.
            posGlyph (tuple): The position of the glyph as a tuple of (x,y) values.
            cellSize (tuple): The size of the glyph cell as a tuple of width and height.
            scale (float): The scale factor of the glyph.

        '''
        x0, y0 = posCell
        x,  y  = posGlyph

        yValues = [
            y,
            y + glyph.font.info.xHeight * scale,
            y + glyph.font.info.ascender * scale,
            y + glyph.font.info.capHeight * scale,
            y - abs(glyph.font.info.descender) * scale,
        ]

        ctx.save()
        ctx.stroke(*self.vMetricsColor)
        ctx.strokeWidth(self.vMetricsStrokeWidth)
        self._setLineDash(self.vMetricsDash, self.vMetricsStrokeWidth)

        for y in yValues:
            ctx.line((x0, y), (x0 + cellSize, y))

        ctx.restore()



[docs]    def drawGlyphShape(self, glyph, posCell, cellSize): # glyphSets
        '''
        Draw a glyph shape in a cell.

        Args:
            glyph (RGlyph): A glyph object.
            posCell (tuple): The position of the glyph cell as a tuple of (x,y) values.
            cellSize (tuple): The size of the glyph cell as a tuple of width and height.

        '''
        fontUPM  = glyph.font.info.unitsPerEm
        posScale = self.getGlyphPosScale(glyph, posCell, cellSize, fontUPM)

        if posScale is None:
            return

        posGlyph, scale = posScale

        if self.vMetricsDraw:
            self.drawGlyphVMetrics(glyph, posCell, posGlyph, cellSize, scale)

        if self.marginsDraw:
            self.drawGlyphMargins(glyph, posCell, posGlyph, cellSize, scale)

        ctx.save()

        if self.glyphFillDraw:
            ctx.fill(*self.glyphFillColor)
        else:
            ctx.fill(None)

        if self.glyphStrokeDraw:
            ctx.stroke(*self.glyphStrokeColor)
        else:
            ctx.stroke(None)

        ctx.translate(*posGlyph)
        ctx.scale(scale)
        ctx.drawGlyph(glyph)
        ctx.restore()

        '''

[docs]    def makeProof(self):
        '''
        Make a PDF proof with the current settings.

        '''
        if not self.font:
            return

        pageWidth, pageHeight = self.pageSize

        w, h = self.pageAreaWidth, self.pageAreaHeight

        fontName   = '%s %s' % (self.font.info.familyName, self.font.info.styleName)
        glyphOrder = self.font.glyphOrder
        cellSize   = self.getCellSize()
        pageCount  = self.pageAmount

        # start new drawing
        ctx.newDrawing()

        for glyphIndex, glyphName in enumerate(glyphOrder):

            # start new page
            if glyphIndex == 0 or y < self.pageMargin:

                if pageCount == 0:
                    break

                x = self.xLeft
                y = self.pageAreaTop - cellSize

                ctx.newPage(pageWidth, pageHeight)

                if self.pageGridDraw:
                    self.drawPageGrid()

                if self.fontNameDraw:
                    self.drawFontName()

                if pageCount is not None:
                    pageCount -= 1

            # draw glyph cell
            self.drawGlyphCell(glyphName, (x, y), cellSize)

            # translate-x
            x += cellSize + self.cellMargin

            # translate-y
            if x > pageWidth - cellSize - self.pageMargin:
                x = self.pageMargin
                y -= (cellSize + self.cellMargin)

        # draw page info

        if self.pageNumberDraw:
            self.drawPageNumbers()

        if self.pageDateTimeDraw:
            self.drawDateTime()
