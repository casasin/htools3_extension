    '''

    @property
    def pageSize(self):
        '''
        The page size as a tuple of width and height.

        '''
        return self.__pageSize

    @pageSize.setter
    def pageSize(self, value):
        '''
        Set the page size.

        '''
        if type(value) is tuple:
            self.__pageSize = value

        else:
            assert type(value) is str
            ctx.newDrawing()
            ctx.newPage(value)
            self.__pageSize = ctx.width(), ctx.height()

    @property
    def layers(self):
        '''
        A list of layers in the font.

        '''
        if not len(self.__layers) and self.font:
            return [self.font.defaultLayer.name]
        else:
            return self.__layers

    @layers.setter
    def layers(self, layerNames):
        self.__layers = layerNames

    @property
    def fontName(self):
        '''
        The name of the current font.

        '''
        familyName = self.font.info.familyName
        styleName  = self.font.info.styleName
        if self.font and len(familyName) and len(styleName):
            return '%s %s' % (self.font.info.familyName, self.font.info.styleName)
        else:
            return 'Untitled'

    @property
    def textTop(self):
        '''
        A boolean indicating if the proof has a top line.

        '''
        return self.fontNameDraw or self.pageNumberDraw

    @property
    def textBottom(self):
        '''
        A boolean indicating if the proof has a bottom line.

        '''
        return self.pageDateTimeDraw

    @property
    def captionAreaHeight(self):
        '''
        The height of the caption area.

        '''
        return self.captionLineHeight + self.pageMargin

    @property
    def xLeft(self):
        '''
        Left edge of the page area.

        '''
        return self.pageMargin

    @property
    def xRight(self):
        '''
        Right edge of the page area.

        '''
        pageWidth, pageHeight = self.pageSize
        return pageWidth - self.pageMargin

    @property
    def yTop(self):
        '''
        Top edge of the page area.

        '''
        pageWidth, pageHeight = self.pageSize
        return pageHeight - self.pageMargin

    @property
    def yBottom(self):
        '''
        Bottom edge of the page area.

        '''
        return self.pageMargin

    @property
    def pageAreaWidth(self):
        '''
        The width of the page area (page width minus lateral margins).

        '''
        pageWidth, pageHeight = self.pageSize
        return pageWidth  - self.pageMargin * 2

    @property
    def pageAreaHeight(self):
        '''
        The height of the page content area (page height minus vertical margins and top/bottom captions).

        '''
        pageWidth, pageHeight = self.pageSize

        areaHeight = pageHeight - self.pageMargin * 2

        if self.textTop:
            areaHeight -= self.captionAreaHeight

        if self.textBottom:
            areaHeight -= self.captionAreaHeight

        return areaHeight

    @property
    def pageAreaTop(self):
        '''
        Top edge of the page content area.

        '''
        y = self.yTop

        if self.textTop:
            y -= self.captionAreaHeight

        return y

    @property
    def pageAreaBottom(self):
        '''
        Bottom edge of the page content area.

        '''
        y = self.yBottom

        if self.textBottom:
            y += self.captionAreaHeight

        return y

    @property
    def pdfData(self):
        '''
        The PDF data for the current proof.

        '''
        return ctx.pdfImage()

[docs]    def getGlyphMetrics(self, glyph):
        '''
        Get metrics info for the given glyph.

        Args:
            glyph (RGlyph): A glyph object.

        Returns:
            The formatted width, left margin and right margin of the glyph.

        '''
        width = glyph.width
        left  = glyph.leftMargin
        right = glyph.rightMargin

        if type(width) is float and not self.glyphMetricsRound:
            width = '%.2f' % width
        elif width is not None:
            width = '%s' % int(width)

        if type(left) is float and not self.glyphMetricsRound:
            left = '%.2f' % left
        elif left is not None:
            left = '%s' % int(left)

        if type(right) is float and not self.glyphMetricsRound:
            right = '%.2f' % right
        elif right is not None:
            right = '%s' % int(right)

        return width, left, right



[docs]    def drawFontName(self):
        '''
        Draw the font name in the current page.

        '''
        w = self.pageAreaWidth
        h = self.captionLineHeight

        x = self.xLeft
        y = self.yTop - h

        ctx.save()
        ctx.fill(*self.captionFillColor)
        ctx.fontSize(self.captionFontSize)
        ctx.lineHeight(self.captionLineHeight)
        ctx.font(self.captionFont)
        ctx.textBox(self.fontName, (x, y, w, h), align='left')
        ctx.restore()



[docs]    def drawPageNumbers(self):
        '''
        Draw page numbers in all pages.

        '''
        w = self.pageAreaWidth
        h = self.captionLineHeight

        x = self.xLeft
        y = self.yTop - h

        ctx.save()
        pages = ctx.pages()
        for i, page in enumerate(pages):
            with page:
                txt = '%s/%s' % (i+1, len(pages))
                ctx.fill(*self.captionFillColor)
                ctx.fontSize(self.captionFontSize)
                ctx.lineHeight(self.captionLineHeight)
                ctx.font(self.captionFont)
                ctx.textBox(txt, (x, y, w, h), align='right')
        ctx.restore()



[docs]    def drawPageGrid(self):
        '''
        Draw page grid in the current page.

        '''
        pageWidth, pageHeight = self.pageSize

        ctx.save()
        ctx.fill(None)
        ctx.stroke(*self.pageGridColor)

        ctx.line((self.xLeft,  0), (self.xLeft,  pageHeight))
        ctx.line((self.xRight, 0), (self.xRight, pageHeight))

        ctx.line((0, self.yBottom), (pageWidth, self.yBottom))
        ctx.line((0, self.yTop),    (pageWidth, self.yTop))

        if self.textTop:
            ctx.line((0, self.pageAreaTop), (pageWidth, self.pageAreaTop))

        if self.textBottom:
            ctx.line((0, self.pageAreaBottom), (pageWidth, self.pageAreaBottom))

        ctx.restore()

                ctx.font(self.captionFont)
                ctx.textBox(now, (x, y, w, h), align='center')

        ctx.restore()

[docs]    def savePDF(self, folder, fileName=None, viewPDF=False):
        '''
        Save proof to PDF file.

        '''
        # get file name
        if fileName:
            fileName = '%s_%s' % (fileName, self.proofName)
        else:
            fileName = '%s_%s' % (self.fontName.replace(' ', '-'), self.proofName)

        # get date & time
        now = str(datetime.now())
        now = now.replace('-', '').replace(':', '').replace('.', '_').replace(' ', '_')

        # make PDF path
        pdfPath = os.path.join(folder, '%s_%s.pdf' % (fileName, now))

        # save PDF to file
        ctx.saveImage(pdfPath, multipage=True)

        # open pdf in Preview
        if viewPDF and os.path.exists(pdfPath):
            openPDF(pdfPath)
