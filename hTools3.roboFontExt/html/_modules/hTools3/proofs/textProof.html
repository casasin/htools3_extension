    @property
    def psFontName(self):
        '''
        The PostScript name of the font.

        '''
        if not self.font:
            return
        fontName = self.font.info.postscriptFontName
        if fontName is None:
            fontName = '%s-%s' % (self.font.info.familyName, self.font.info.styleName)
            fontName = fontName.replace(' ', '')
        return fontName

        '''

            sampleName (str): The name of the sample.

        '''
        w = self.pageAreaWidth
        h = self.captionLineHeight

        x = self.xLeft
        y = self.yTop - h

        ctx.save()
        ctx.fill(*self.captionFillColor)
        ctx.fontSize(self.captionFontSize)
        ctx.lineHeight(self.captionLineHeight)
        ctx.font(self.captionFont)
        ctx.textBox(sampleName, (x, y, w, h), align='center')
        ctx.restore()



[docs]    def drawSampleTextPage(self, txt, sampleName):
        '''
        Draw sample text and proof items on the current page.

        Args:
            txt (str): The sample text.
            sampleName (str): The name of the sample.

        '''
        ctx.newPage(*self.pageSize)

        if self.pageGridDraw:
            self.drawPageGrid()

        if self.fontNameDraw:
            self.drawFontName()

        if self.sampleNameDraw:
            self.drawSampleName(sampleName)

        x = self.xLeft
        y = self.pageAreaBottom

        w = self.pageAreaWidth
        h = self.pageAreaHeight

        ctx.save()
        ctx.hyphenation(self.textHyphenation)
        ctx.lineCap('round')
        ctx.lineJoin('round')

        T = ctx.FormattedString()

        if self.textFillDraw:
            T.fill(*self.textFillColor)
        else:
            T.fill(None)

        if self.textStrokeDraw:
            T.stroke(*self.textStrokeColor)
            T.strokeWidth(self.textStrokeWidth)
        else:
            T.stroke(None)

        T.font(self.psFontName)
        T.fontSize(self.textFontSize)
        T.lineHeight(self.textFontSize * self.textLineHeight)
        T.tracking(self.textTracking)
        T.align(self.textAlignment)

        allFeatures = ctx.listOpenTypeFeatures(self.psFontName)
        cmd = 'T.openTypeFeatures('
        for i, feaName in enumerate(allFeatures):
            cmd += '%s=%s' % (feaName, feaName in self.fontFeatures)
            if i < len(allFeatures):
                cmd += ', '
        cmd += ')'
        exec(cmd)

        T.append(txt)

        # draw vertical metrics
        if self.vMetricsDraw:
            ctx.save()
            ctx.stroke(*self.vMetricsColor)
            ctx.strokeWidth(self.vMetricsStrokeWidth)
            ctx.font(self.psFontName)
            ctx.fontSize(self.textFontSize)
            ctx.lineHeight(self.textFontSize * self.textLineHeight)
            yValues = set([
                0,
                ctx.fontDescender(),
                ctx.fontXHeight(),
                ctx.fontCapHeight(),
                ctx.fontAscender(),
            ])
            for baselinePos in ctx.textBoxBaselines(T, (x, y, w, h)):
                yBaseline = baselinePos[1]
                with ctx.savedState():
                    ctx.translate(x, yBaseline)
                    for yValue in yValues:
                        ctx.line((0, yValue), (w, yValue))
            ctx.restore()

        # draw text
        txt = ctx.textBox(T, (x, y, w, h))

        ctx.restore()

        return txt
