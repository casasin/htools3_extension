[docs]def rasterize(glyph, cellSize=50, xMin=None, yMin=None, xMax=None, yMax=None):
    """
    Slice the glyph into a grid of the given cell size.

    Args:
        cellSize (int): The size of the grid cells.
        xMin (int): The minimum x-value to use when scanning the glyph.
        yMin (int): The minimum y-value to use when scanning the glyph.
        xMax (int): The maximum x-value to use when scanning the glyph.
        yMax (int): The maximum y-value to use when scanning the glyph.

    Returns:
        A list of lists containing bool values that indicate the
    black (``True``) or white (``False``) value of that particular cell.
    These lists are arranged from top to bottom of the glyph and proceed
    from left to right.

    .. note::

        Ported from ``robofab.objects.objectsBase.BaseGlyph``. Not used by the *Rasterizer* object.

    """

    if xMin is None or yMin is None or xMax is None or yMax is None:
        _xMin, _yMin, _xMax, _yMax = glyph.bounds
        if xMin is None:
            xMin = _xMin
        if yMin is None:
            yMin = _yMin
        if xMax is None:
            xMax = _xMax
        if yMax is None:
            yMax = _yMax

    hitXMax = False
    hitYMin = False
    xSlice = 0
    ySlice = 0
    halfCellSize = cellSize / 2.0
    bitmap = []

    while not hitYMin:
        bitmap.append([])
        yScan = -(ySlice * cellSize) + yMax - halfCellSize
        if yScan < yMin:
            hitYMin = True
        while not hitXMax:
            xScan = (xSlice * cellSize) + xMin - halfCellSize
            if xScan > xMax:
                hitXMax = True
            test = glyph.pointInside((xScan, yScan))
            if test:
                bitmap[-1].append(True)
            else:
                bitmap[-1].append(False)
            xSlice = xSlice + 1
        hitXMax = False
        xSlice = 0
        ySlice = ySlice + 1

    return bitmap



[docs]def drawElement(glyph, size, type='rect', roundness=0.8):
    '''
    Set the shape of the element glyph in the font.

    Args:
        size (tuple): The size of the element shape.
        type (str): The type of the element shape: ``rect``, ``oval`` or ``element``.
        roundness (float): The roundness of the element shape. Used only when the element type is ``element``.

    '''
    w, h = size

    # get a pen to draw in the glyph
    pen = glyph.getPen()

    # draw element in glyph
    if type == 'oval':
        oval(pen, 0, 0, w, h)
    elif type == 'rect':
        rect(pen, 0, 0, w, h)
    else:
        element(pen, 0, 0, w, h, ratio=roundness)

    # set glyph width
    glyph.width = w



[docs]def randomizeElements(glyph, elementSize, randSize):
    '''
    Randomize the size of element shapes in the current glyph.

    '''
    for e in glyph.components:

        # calculate size
        randMin = int(float(randSize[0]) * 100)
        randMax = int(float(randSize[1]) * 100)
        s = random.randint(randMin, randMax) * 0.01
        sx = elementSize[0] * s
        sy = elementSize[1] * s

        # calculate position
        x, y = e.offset
        x += ((elementSize[0] - sx) * 0.5)
        y += ((elementSize[1] - sy) * 0.5)

        # transform element
        e.offset = (x, y)
        e.scale = (s, s)



[docs]def getElementSize(font, glyphName='_element'):
    '''
    Get a font's element size from its bounding box.

    '''
    xMin, yMin, xMax, yMax = font[glyphName].bounds
    w = xMax - xMin
    h = yMax - yMin
    return w, h

[docs]    def scan(self):
        '''
        Scan the glyph and store the resulting on/off bits in the glyph lib.

        '''
        gridX = gridY = self.elementSpace

        xMin, xMax = 0, self.glyph.width
        yMin, yMax = self.yMin, self.yMax

        # round values to grid
        xMin = (xMin // gridX) * gridX
        yMin = (yMin // gridY) * gridY
        xMax = (xMax // gridX) * gridX
        yMax = (yMax // gridY) * gridY

        # make grid lines
        xValues = range(xMin, xMax, gridX)
        yValues = range(yMin, yMax, gridY)

        # scan lines
        lines = {}
        for y in reversed(yValues):
            lineNumber = y // gridY
            bits = []
            for x in xValues:
                ptx = x + gridX * 0.5
                pty = y + gridY * 0.5
                if self.glyph.pointInside((ptx, pty)):
                    bits.append(1,)
                else:
                    bits.append(0,)
            lines[str(lineNumber)] = bits

        # store bits in the glyph lib
        self.glyph.lib[self.libKeyCoordinates] = lines



[docs]    def print(self, black="#", white="-"):
        '''
        Print glyph bits as ASCII text.

        '''
        lineNumbers = [int(L) for L in self.bits.keys()]
        left, right = [int(m) for m in self.margins]
        for i in reversed(sorted(lineNumbers)):
            L = str(i)
            lineBits = [str(b) for b in self.bits[L]]
            lineBits = '%s%s%s' % (left * '0', ''.join(lineBits), right * '0')
            lineBits = lineBits.replace('1', black).replace('0', white)
            lineBits = ' '.join(list(lineBits))
            print('%+03d' % i, lineBits)



[docs]    def draw(self, targetGlyph=None, components=True):
        '''
        Render scanned bits into destination glyph using components.

        '''

        # define destination glyph
        if targetGlyph == None:
            targetGlyph = self.glyph

        # destination cannot be element itself
        if targetGlyph.name == self.elementGlyph:
            return

        # get parameters
        gridX  = gridY = self.elementSpace
        sizeX  = sizeY = self.elementSize
        scaleX = sizeX / float(gridX)
        scaleY = sizeY / float(gridY)

        # prepare destination glyph
        targetGlyph.prepareUndo('rasterizer')
        targetGlyph.clear()

        # prepare lines
        lineNumbers = reversed(sorted(self.bits.keys()))

        # draw bits from lib
        for lineNumber in lineNumbers:
            bitCount = 0
            for bit in self.bits[lineNumber]:
                if bit == 1:
                    x = bitCount * gridX
                    y = int(lineNumber) * gridY
                    if components:
                        targetGlyph.appendComponent(self.elementGlyph, offset=(x, y), scale=(scaleX, scaleY))
                    else:
                        pen = targetGlyph.getPen()
                        oval(pen, x, y, sizeX, sizeY)
                else:
                    pass
                bitCount = bitCount + 1

        # targetGlyph.rightMargin = round(rightMargin) * gridX

        targetGlyph.performUndo()
